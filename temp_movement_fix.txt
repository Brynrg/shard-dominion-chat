    private tileCache: Map<string, Position> = new Map();

    setPathfinding(pathfinding: PathfindingSystem): void {
        this.pathfinding = pathfinding;
    }

    setPlanetAgitationSystem(system: any): void {
        this.planetAgitationSystem = system;
    }

    // Configure pathfinder with current tiles
    configurePathfinder(tiles: Tile[][]): void {
        this.pathfinding.setTiles(tiles);
    }

    // Cache waypoints for smoother movement
    cacheWaypoint(unitId: string, tile: Position): void {
        this.tileCache.set(unitId, tile);
    }

    getWaypoint(unitId: string): Position | undefined {
        return this.tileCache.get(unitId);
    }
